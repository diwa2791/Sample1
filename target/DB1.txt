private RowDiffDTO findMatchingRow(
        List<RowDiffDTO> rows,
        Map<String, Object> requestPk) {

    if (rows == null || requestPk == null)
        return null;

    outer:
    for (RowDiffDTO row : rows) {

        Map<String, Object> rowPk = row.getPrimaryKey();

        if (rowPk.size() != requestPk.size())
            continue;

        for (Map.Entry<String, Object> entry : rowPk.entrySet()) {

            Object reqVal = requestPk.get(entry.getKey());

            if (reqVal == null)
                continue outer;

            if (!String.valueOf(entry.getValue())
                    .equals(String.valueOf(reqVal)))
                continue outer;
        }

        return row; // match found
    }

    return null;
}

public RowDiffDTO getRowDetails(
        String cacheId,
        Map<String,Object> pk,
        String tab) {

    TableDiffResponseDTO diff = cache.get(cacheId).data;

    List<RowDiffDTO> list =
            switch (tab) {
                case "test" -> diff.onlyInTestRows;
                case "prod" -> diff.onlyInProdRows;
                default     -> diff.modifiedRows;
            };

    return findMatchingRow(list, pk);
}
