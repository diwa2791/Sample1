async function loadRow(btn) {

    const expandRow = btn.closest('tr').nextElementSibling;
    const container = expandRow.querySelector('.expand-content');

    if (expandRow.style.display === 'table-row') {
        expandRow.style.display = 'none';
        return;
    }

    expandRow.style.display = 'table-row';
    container.innerHTML = "Loading...";

    const pk = btn.dataset.pk;

    const res = await fetch(
        `/compare/row?cacheId=${cacheId}&pk=${encodeURIComponent(pk)}&tab=${tab}`
    );

    const data = await res.json();

    container.innerHTML = buildExpandHtml(data);
}

function buildExpandHtml(data) {

    let html = `<table class="inner-table">
        <tr>
            <th>Column</th>
            <th>Test</th>
            <th>Prod</th>
        </tr>`;

    Object.keys(data.testRow || {}).forEach(key => {

        html += `
        <tr>
            <td>${key}</td>
            <td>${data.testRow[key] ?? ''}</td>
            <td>${data.prodRow[key] ?? ''}</td>
        </tr>`;
    });

    html += "</table>";

    return html;
}
